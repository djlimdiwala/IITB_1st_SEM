Here are formulas for gaussian distribution.
\begin{equation}
	F (x) = \frac{1}{\sigma \sqrt {2\pi}}e^{\frac{ - \left( {x - \mu } \right)^2}{2\sigma^2}}
\end{equation}
where,
\begin{equation}
	\mu = \frac{1}{N}\sum_{i = 1}^{N}x_i
\end{equation}
\begin{equation}
	\sigma^2 = \frac{1}{N}\sum_{i=1}^{N}(x_i - \mu)^2
\end{equation}
